function getMiddle(s){
    let arr = s.split('');
    let start = 0;
    let end = arr.length - 1;
    let middle = Math.floor((start + end) / 2);
    if(arr.length % 2 == 0) {
        return `${arr[middle]}${arr[middle + 1]}`
    } else {
        let middle = Math.floor((start + end) / 2);
        return `${arr[middle]}`
    }
}
